# DX Provider

Interact with the DX Web API and declaratively manage Service Cloud resources.

## Example Usage

```terraform
terraform {
  required_providers {
    dx = {
      source = "registry.terraform.io/local/dx"
    }
  }
}

provider "dx" {
  # Define your Web API token here, or set `DX_WEB_API_TOKEN` in your environment.
  #
  # To manage scorecards, the token must have the following scopes:
  #
  # - scorecards:read
  # - scorecards:write
  #
  api_token = "SnrHqiwpprB8FvzJ2V3EXLmvJWfeqNR9wYgP"
}

resource "dx_entity_type" "db-resource" {
  identifier  = "db-resource"
  name        = "DB Resource"
  description = "A database resource"
}

# Example: Create an entity type with properties and aliases
resource "dx_entity_type" "repository" {
  identifier  = "repository"
  name        = "Repository"
  description = "A source code repository"

  properties = [
    {
      identifier = "team"
      name       = "Owning Team"
      type       = "multi_select"
      visibility = "visible"
      options    = ["platform", "data", "product", "infrastructure"]
    },
    {
      identifier = "language"
      name       = "Primary Language"
      type       = "text"
      visibility = "visible"
    },
    {
      identifier = "tier"
      name       = "Service Tier"
      type       = "multi_select"
      visibility = "hidden"
      options    = ["tier_1", "tier_2", "tier_3"]
    }
  ]

  aliases = {
    "github_repo" = true
  }
}
```
